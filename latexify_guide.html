LaTeXify Matplotlib Figures: A Complete Guide

Published

May 23, 2025

Overview

This guide demonstrates how to create publication-ready matplotlib figures using the latexify function developed by Prof. Nipun Batra. This tool helps format matplotlib figures to match LaTeX document styling, making them suitable for academic papers and professional publications.

The LaTeXify Code

#Original Author: Prof. Nipun Batra # nipunbatra.github.io
from math import sqrt
import matplotlib.pyplot as plt
import numpy as np

SPINE_COLOR = 'gray'

def latexify(fig_width=None, fig_height=None, columns=1):
    """Set up matplotlib's RC params for LaTeX plotting.
    Call this before plotting a figure.
    
    Parameters
    ----------
    fig_width : float, optional, inches
    fig_height : float, optional, inches  
    columns : {1, 2}
        Number of columns in your LaTeX document
    """
    # code adapted from http://www.scipy.org/Cookbook/Matplotlib/LaTeX_Examples
    # Width and max height in inches for IEEE journals taken from
    # computer.org/cms/Computer.org/Journal%20templates/transactions_art_guide.pdf
    
    assert(columns in [1,2])
    
    if fig_width is None:
        fig_width = 3.39 if columns==1 else 6.9  # width in inches
        
    if fig_height is None:
        golden_mean = (sqrt(5)-1.0)/2.0  # Aesthetic ratio
        fig_height = fig_width*golden_mean  # height in inches
        
    MAX_HEIGHT_INCHES = 8.0
    if fig_height > MAX_HEIGHT_INCHES:
        print(f"WARNING: fig_height too large: {fig_height} so will reduce to {MAX_HEIGHT_INCHES} inches.")
        fig_height = MAX_HEIGHT_INCHES
    
    params = {
        'backend': 'ps',
        'text.latex.preamble': r'\usepackage{gensymb}',
        'axes.labelsize': 8,      # fontsize for x and y labels (was 10)
        'axes.titlesize': 8,
        'font.size': 8,           # was 10
        'legend.fontsize': 8,     # was 10
        'xtick.labelsize': 8,
        'ytick.labelsize': 8,
        'text.usetex': True,
        'figure.figsize': [fig_width, fig_height],
        'font.family': 'serif'
    }
    
    plt.rcParams.update(params)

def format_axes(ax):
    """Format axes to have a clean, publication-ready appearance."""
    for spine in ['top', 'right']:
        ax.spines[spine].set_visible(False)
        
    for spine in ['left', 'bottom']:
        ax.spines[spine].set_color(SPINE_COLOR)
        ax.spines[spine].set_linewidth(0.5)
        
    ax.xaxis.set_ticks_position('bottom')
    ax.yaxis.set_ticks_position('left')
    
    for axis in [ax.xaxis, ax.yaxis]:
        axis.set_tick_params(direction='out', color=SPINE_COLOR)
        
    return ax

1. Automatic Sizing

  • Single column: 3.39 inches width (IEEE standard)
  • Double column: 6.9 inches width

Basic Usage Examples

Example 1: Simple Line Plot

import numpy as np
import matplotlib.pyplot as plt

# Set up LaTeX formatting for single column
latexify(columns=1)

# Create data
x = np.linspace(0, 2*np.pi, 100)
y = np.sin(x)

# Create plot
fig, ax = plt.subplots()
ax.plot(x, y, 'b-', linewidth=1.5, label=r'$\sin(x)$')
ax.set_xlabel(r'$x$ (radians)')
ax.set_ylabel(r'$\sin(x)$')
ax.legend()

# Format axes
format_axes(ax)

# Save as PDF for LaTeX
plt.savefig('sine_plot.pdf', format='pdf', dpi=300, bbox_inches='tight')
plt.show()

Example 2: Multi-panel Figure

# Set up for double column figure
latexify(fig_width=6.9, fig_height=3, columns=2)

# Create subplots
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(6.9, 3))

# Left panel - Exponential decay
x = np.linspace(0, 5, 100)
y1 = np.exp(-x)
ax1.plot(x, y1, 'r-', linewidth=1.5)
ax1.set_xlabel(r'Time ($t$)')
ax1.set_ylabel(r'$e^{-t}$')
ax1.set_title('Exponential Decay')
format_axes(ax1)

# Right panel - Power law
x2 = np.logspace(0, 2, 100)
y2 = x2**(-2)
ax2.loglog(x2, y2, 'g-', linewidth=1.5)
ax2.set_xlabel(r'$x$')
ax2.set_ylabel(r'$x^{-2}$')
ax2.set_title('Power Law')
format_axes(ax2)

plt.tight_layout()
plt.savefig('multi_panel.pdf', format='pdf', dpi=300, bbox_inches='tight')
plt.show()

Example 3: Scientific Data Visualization

# Data analysis example
latexify(columns=1)

# Generate sample data
np.random.seed(42)
temperatures = np.random.normal(25, 5, 100)  # Temperature in Celsius
measurements = 2.5 * temperatures + np.random.normal(0, 2, 100)

fig, ax = plt.subplots()

# Scatter plot with trend line
ax.scatter(temperatures, measurements, alpha=0.6, s=20, color='blue', label='Data')

# Fit and plot trend line
z = np.polyfit(temperatures, measurements, 1)
p = np.poly1d(z)
ax.plot(temperatures, p(temperatures), 'r--', linewidth=1.5, 
        label=rf'Fit: $y = {z[0]:.2f}x + {z[1]:.2f}$')

ax.set_xlabel(r'Temperature (\degree C)')
ax.set_ylabel(r'Measurement (units)')
ax.set_title('Temperature vs Measurement')
ax.legend()

format_axes(ax)
plt.savefig('scientific_plot.pdf', format='pdf', dpi=300, bbox_inches='tight')
plt.show()

Common Issues and Solutions

1. LaTeX Not Found

# If LaTeX is not installed, disable LaTeX rendering
def latexify_no_latex(fig_width=None, fig_height=None, columns=1):
    # Same function but with 'text.usetex': False
    params['text.usetex'] = False
    plt.rcParams.update(params)

2. Font Size Too Small

# Increase font sizes for presentations
def latexify_presentation(fig_width=None, fig_height=None):
    params = {
        'axes.labelsize': 14,
        'axes.titlesize': 16,
        'font.size': 12,
        'legend.fontsize': 12,
        'xtick.labelsize': 12,
        'ytick.labelsize': 12,
    }
    plt.rcParams.update(params)

3. High DPI for Print

# Save with high resolution for print
plt.savefig('figure.pdf', format='pdf', dpi=300, bbox_inches='tight')

Best Practices

  1. Always call latexify() before creating figures
  2. Use format_axes() on all axes for consistency
  3. Save as PDF format for LaTeX documents (modern LaTeX supports PDF natively)
  4. Use raw strings (r'') for LaTeX expressions
  5. Test your figures in the actual LaTeX document
  6. Keep line weights between 1.0-2.0 for clarity
  7. Use bbox_inches='tight' to remove extra whitespace
  8. Use 300 DPI for high-quality output

Conclusion

The latexify function provides a simple way to create publication-quality figures that integrate seamlessly with LaTeX documents. By following the examples and best practices in this guide, you can produce professional-looking visualizations for your research papers and presentations.
